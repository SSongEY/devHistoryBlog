```java
public static ListNode reverseList(ListNode head) {
    ListNode cursor = null;
    
    while (head != null) {
        ListNode nextNode = head.next;
        head.next = cursor;
        cursor = head;
        head = nextNode;
    }
    
    return cursor;
}
```
